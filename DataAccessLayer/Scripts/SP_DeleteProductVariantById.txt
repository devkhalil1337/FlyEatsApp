ALTER PROCEDURE [dbo].[SP_DeleteProductVariantById]
(
	@ProductId int
	
)
AS  
BEGIN   
Declare @status int  
If EXISTS(Select Count(*) from ProductVariants where ProductId = @ProductId)
   update  ProductVariants SET IsDeleted=1 where ProductId = @ProductId
   Begin
   SET @status = 1;
    return @status;
 End
END
